        // Game management

        function loadGame(gameFile) {

            const mainMenu = document.getElementById('main-menu');

            const gameContainer = document.getElementById('game-container');

            const gameFrame = document.getElementById('game-frame');

            const homeBtn = document.getElementById('homeBtn');


            // Hide main menu and show game container

            mainMenu.style.display = 'none';

            gameContainer.classList.add('active');

            homeBtn.style.display = 'block';


            // Load game in iframe

            gameFrame.src = gameFile;


            // Scroll to top

            window.scrollTo(0, 0);

        }


        function showMainMenu() {

            const mainMenu = document.getElementById('main-menu');

            const gameContainer = document.getElementById('game-container');

            const gameFrame = document.getElementById('game-frame');

            const homeBtn = document.getElementById('homeBtn');


            // Show main menu and hide game container

            mainMenu.style.display = 'block';

            gameContainer.classList.remove('active');

            homeBtn.style.display = 'none';


            // Unload game from iframe

            gameFrame.src = '';


            // Scroll to top

            window.scrollTo(0, 0);

        }


        // Handle iframe errors

        document.getElementById('game-frame').addEventListener('load', function() {

            console.log('Game loaded successfully');

        });


        // Premium modal functions

        function showPremiumModal() {

            event.stopPropagation(); // Prevent card click event

            const modal = document.getElementById('premium-modal');

            modal.classList.add('active');

        }


        function closePremiumModal() {

            const modal = document.getElementById('premium-modal');

            modal.classList.remove('active');

        }


        // Keyboard navigation

        document.addEventListener('keydown', function(e) {

            // ESC key to return to main menu or close modal

            if (e.key === 'Escape') {

                const modal = document.getElementById('premium-modal');

                if (modal.classList.contains('active')) {

                    closePremiumModal();

                } else if (document.getElementById('homeBtn').style.display !== 'none') {

                    showMainMenu();

                }

            }

        });
